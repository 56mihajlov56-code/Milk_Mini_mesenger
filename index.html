import os
import sys
import winreg
import psutil
import shutil
import hashlib
import ctypes
from datetime import datetime

# --- Антивирус от L1 team ---
# Emergency Recovery Script v1.2

WHITE_LIST_PROCESSES = [
    "explorer.exe", "svchost.exe", "lsass.exe", "services.exe",
    "wininit.exe", "smss.exe", "csrss.exe", "taskhostw.exe",
    "runtimebroker.exe", "winlogon.exe"
]

SUSPICIOUS_PATHS = [
    os.environ.get('TEMP', ''),
    os.environ.get('APPDATA', ''),
    os.environ.get('LOCALAPPDATA', ''),
    os.path.join(os.environ.get('SystemDrive', 'C:'), '\\'),
]

EXTENSIONS = ('.exe', '.scr', '.vbs', '.ps1', '.bat', '.cmd')

class L1Antivirus:
    def __init__(self):
        self.deleted_files = []
        self.cleaned_registry = []
        self.scan_start_time = datetime.now()

    def log(self, message, type="INFO"):
        prefix = f"[{type}]"
        print(f"{datetime.now().strftime('%H:%M:%S')} {prefix} {message}")

    def is_admin(self):
        try:
            return ctypes.windll.shell32.IsUserAnAdmin()
        except:
            return False

    def scan_registry(self):
        self.log("Сканирование веток автозагрузки реестра...")
        keys = [
            (winreg.HKEY_LOCAL_MACHINE, r"Software\Microsoft\Windows\CurrentVersion\Run"),
            (winreg.HKEY_LOCAL_MACHINE, r"Software\Microsoft\Windows\CurrentVersion\RunOnce"),
            (winreg.HKEY_CURRENT_USER, r"Software\Microsoft\Windows\CurrentVersion\Run"),
            (winreg.HKEY_CURRENT_USER, r"Software\Microsoft\Windows\CurrentVersion\RunOnce")
        ]

        for root, path in keys:
            try:
                with winreg.OpenKey(root, path, 0, winreg.KEY_ALL_ACCESS) as key:
                    i = 0
                    while True:
                        try:
                            name, value, _ = winreg.EnumValue(key, i)
                            # Интеллектуальный фильтр
                            is_suspicious = False
                            val_lower = value.lower()
                            
                            # Проверка на белый список и пути
                            if not any(proc in val_lower for proc in WHITE_LIST_PROCESSES):
                                if any(p.lower() in val_lower for p in SUSPICIOUS_PATHS) or ":" not in value:
                                    is_suspicious = True
                            
                            if is_suspicious:
                                self.log(f"Удаление подозрительного ключа реестра: {name} ({value})", "WARN")
                                winreg.DeleteValue(key, name)
                                self.cleaned_registry.append(f"{path}\\{name}")
                                continue # Не инкрементируем индекс, так как элемент удален
                            i += 1
                        except OSError:
                            break
            except Exception as e:
                self.log(f"Доступ к разделу {path} ограничен", "ERROR")

    def heuristic_scan(self):
        self.log("Запуск эвристического анализа файлов (AI-подобная логика)...")
        for drive in [f"{d}:\\" for d in "CDEFGHIJKLMNOPQRSTUVWXYZ" if os.path.exists(f"{d}:\\")]:
            self.log(f"Проверка диска {drive}...")
            
            # Анализ RAT и файлов-пустышек (100MB аномалии)
            size_map = {}
            
            for root, dirs, files in os.walk(drive):
                # Ограничение глубины для WinRE
                if root.count(os.sep) > 8: continue 
                
                for file in files:
                    file_path = os.path.join(root, file)
                    try:
                        # 1. Проверка подозрительных расширений в Temp/AppData
                        if any(sp.lower() in root.lower() for sp in SUSPICIOUS_PATHS):
                            if file.lower().endswith(EXTENSIONS):
                                self.log(f"Обнаружен опасный объект в системной папке: {file}", "CRITICAL")
                                os.remove(file_path)
                                self.deleted_files.append(file_path)
                                continue

                        # 2. Анализ дублей и аномалий (RAT Check)
                        fsize = os.path.getsize(file_path)
                        if fsize > 90 * 1024 * 1024 and fsize < 110 * 1024 * 1024: # Около 100 МБ
                            # Если нашли 3+ одинаковых по размеру файла в странных местах
                            size_map[fsize] = size_map.get(fsize, []) + [file_path]
                            if len(size_map[fsize]) > 3:
                                for path_to_del in size_map[fsize]:
                                    if os.path.exists(path_to_del):
                                        self.log(f"Удаление RAT-клона: {path_to_del}", "CRITICAL")
                                        os.remove(path_to_del)
                                        self.deleted_files.append(path_to_del)

                    except Exception:
                        continue

    def cleanup_processes(self):
        self.log("Анализ активных процессов...")
        for proc in psutil.process_iter(['pid', 'name', 'exe']):
            try:
                exe_path = proc.info['exe']
                if not exe_path: continue
                
                # Если процесс запущен из подозрительного места и не в белом списке
                if any(sp.lower() in exe_path.lower() for sp in SUSPICIOUS_PATHS):
                    if proc.info['name'].lower() not in WHITE_LIST_PROCESSES:
                        self.log(f"Завершение подозрительного процесса: {proc.info['name']} (PID: {proc.info['pid']})", "WARN")
                        proc.kill()
            except (psutil.NoSuchProcess, psutil.AccessDenied):
                continue

    def run(self):
        print("\n" + "="*50)
        print("--- Антивирус от L1 team ---")
        print("="*50 + "\n")
        
        if not self.is_admin():
            self.log("ВНИМАНИЕ: Скрипт запущен без прав администратора!", "ERROR")
        
        self.cleanup_processes()
        self.scan_registry()
        self.heuristic_scan()
        
        duration = datetime.now() - self.scan_start_time
        
        print("\n" + "="*50)
        print("--- L1 team: сканирование завершено ---")
        print(f"Время работы: {duration}")
        print(f"Удалено файлов: {len(self.deleted_files)}")
        print(f"Очищено ключей реестра: {len(self.cleaned_registry)}")
        print("="*50)
        
        if self.deleted_files or self.cleaned_registry:
            print("\nПОДРОБНЫЙ ОТЧЕТ:")
            for f in self.deleted_files: print(f"[-] Файл: {f}")
            for r in self.cleaned_registry: print(f"[-] Реестр: {r}")
        else:
            print("\nУгроз не обнаружено. Система чиста.")

if __name__ == "__main__":
    scanner = L1Antivirus()
    scanner.run()
    input("\nНажмите Enter для выхода...")

Подготовка

Сохраните код в файл antivirus.py на USB-накопитель. Убедитесь, что на целевой системе установлен Python или используйте портативную версию.

02

Вход в WinRE

Загрузитесь с установочного диска Windows -> Восстановление системы -> Поиск и устранение неисправностей -> Командная строка.

03

Выполнение
